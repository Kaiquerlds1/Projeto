CREATE TABLE `estabelecimentos` ( 
	`id` INT NOT NULL AUTO_INCREMENT , 
	`razao` VARCHAR(250) NULL DEFAULT NULL , 
	`fantasia` VARCHAR(250) NULL DEFAULT NULL , 
	`cnpj` VARCHAR(20) NULL DEFAULT NULL , 
	`email` VARCHAR(100) NULL DEFAULT NULL , 
	`endereco` VARCHAR(100) NULL DEFAULT NULL , 
	`cidade` VARCHAR(50) NULL DEFAULT NULL , 
	`estado` VARCHAR(50) NULL DEFAULT NULL , 
	`categoria` VARCHAR(50) NULL DEFAULT NULL ,
	`telefone` VARCHAR(20) NULL DEFAULT NULL , 
	`status` VARCHAR(20) NULL DEFAULT NULL , 
	`agencia` VARCHAR(30) NULL DEFAULT NULL , 
	`conta` VARCHAR(30) NULL DEFAULT NULL , 
	`datacadastro` DATE NULL DEFAULT NULL , 
	PRIMARY KEY (`id`)
) CHARSET=latin1 COLLATE latin1_swedish_ci;
